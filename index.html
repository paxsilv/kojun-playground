Kojun Programming Language

Kojun is an experimental, dynamically typed scripting language that I built as a learning project. The interpreter is written in Python and largely based on the jlox implementation from the book Crafting Interpreters. Try running the examples below or write your own code in the editor. 
let config_found = true;

if (config_found:
    display("Configuration loaded.");
) else (:
    display("Configuration file missing.");
)

var i = 0;

while (i < 10:
    display(i);
    newline();
    i += 1;
)

fun square(x:
    return x * x;
)

fun map(fn, items:
    let N = length(items);
    let new_array = array(nil, N);

    var i = 0;
    while (i < N:
        new_array[i] = fn(items[i]);
        i += 1;
    )

    return new_array;
)

display(map(square, [2, 3, 4, 5, 6, 7]));
